<script>
    function checkOther() {
        let qual = document.getElementById("qualification").value;
        document.getElementById("otherQual").style.display = (qual === "Other") ? "block" : "none";
    }

    function analyze() {
        let name = document.getElementById("name").value;
        let qual = document.getElementById("qualification").value;
        let fullDate = document.getElementById("date").value;

        if (name === "" || qual === "" || fullDate === "") {
            alert("Fill all fields");
            return;
        }

        let year = new Date(fullDate).getFullYear();
        document.getElementById("formPage").style.display = "none";
        document.getElementById("resultPage").style.display = "block";

        document.getElementById("r_name").innerText = name;
        document.getElementById("r_qualification").innerText = qual;
        document.getElementById("r_year").innerText = year;

        let companies = "TCS | Infosys | Wipro | Accenture | Capgemini";
        document.getElementById("companyData").innerText = companies;

        let job = "Data Analyst";

        document.getElementById("naukriLink").href = "https://www.naukri.com/" + job.replace(/ /g, "-") + "-jobs";
        document.getElementById("indeedLink").href = "https://www.indeed.com/jobs?q=" + job.replace(/ /g, "+");
        document.getElementById("founditLink").href = "https://www.foundit.in/srp/results?query=" + job.replace(/ /g, "%20");
        document.getElementById("shineLink").href = "https://www.shine.com/job-search/" + job.replace(/ /g, "-") + "-jobs";
        document.getElementById("timesjobsLink").href = "https://www.timesjobs.com/candidate/job-search.html?txtKeywords=" + job.replace(/ /g, "%20");
        document.getElementById("glassdoorLink").href = "https://www.glassdoor.com/Job/jobs.htm?sc.keyword=" + job.replace(/ /g, "%20");
        document.getElementById("linkedinLink").href = "https://www.linkedin.com/jobs/search/?keywords=" + job.replace(/ /g, "%20");
    }
</script>
